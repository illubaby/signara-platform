// Example 1: Basic usage
function openPicker1() {
    FilePicker.open((path, isDirectory) => {
        document.getElementById('selectedPath1').value = path;
    });
}

// Example 2: With type indicator
function openPicker2() {
    FilePicker.open((path, isDirectory) => {
        document.getElementById('selectedPath2').value = path;
        const badge = document.getElementById('pathType2');
        badge.textContent = isDirectory ? 'Directory' : 'File';
        badge.className = isDirectory ? 'badge badge-success' : 'badge badge-primary';
    });
}

// Example 3: Directory only (custom validation)
function openPicker3() {
    FilePicker.open((path, isDirectory) => {
        if (!isDirectory) {
            alert('Please select a directory, not a file!');
            return;
        }
        document.getElementById('selectedPath3').value = path;
    });
}
